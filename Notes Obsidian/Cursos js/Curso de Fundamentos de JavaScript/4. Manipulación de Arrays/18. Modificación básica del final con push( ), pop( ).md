```javascript
// Methods that modify the original array (Mutability).

// push()

const countries = ['USA', 'Canada', 'UK']
// El push aparte de agregar los nuevos elementos regresa la nueva longitud del arreglo
const newCountries = countries.push('Germany', 'Australia')

console.log(countries)
console.log(newCountries)

// pop()
// El pop elimina el ultimo elemento del arreglo y regresa el elemento eliminado
const removedCountry = countries.pop()

console.log(countries)
console.log(removedCountry)
```

```shell
[ 'USA', 'Canada', 'UK', 'Germany', 'Australia' ]
5
[ 'USA', 'Canada', 'UK', 'Germany' ]
Australia
```
