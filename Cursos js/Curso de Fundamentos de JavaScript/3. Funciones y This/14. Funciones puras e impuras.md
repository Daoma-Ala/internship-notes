```javascript
// Funciones puras
// Caracteristicas: Da una misma entrada y obtener la misma salida
// 			No produce efectos secundarios

// Side Efects -- lo que hace que no sea pura -- Efectos secundarios
// 1. Modificar variables globales
// 2. Modificar parametros
// 3. Solicitudes  HTTP
// 4. Imprimir mensajes en pantalla o en consola
// 5. Manipulación del DOM
// 6. Acceder a la hora o al día

// Funcion pura-- misma entrada y misma salida
function sum (a, b){
	return a+b;
}

// Función impuras -- Efectos secundarios -- 4. Imprimir mensajes en pantalla
function sum (a, b){
	console.log('A: '+a);
	return a+b;
}

// Funcion impura -- 1. Modificacion de variable impura
let total = 0;
function sumWithSideEffect (a){
	total +=a;
	return total;
}

// Funcion pura
function square (x){
	return 	x*x;
}
```
